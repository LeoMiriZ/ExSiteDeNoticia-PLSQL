CREATE TABLE CATEGORIA
(ID_CATEGORIA NUMBER (10,0),
NOME_CATEGORIA VARCHAR (30),
CONSTRAINT ID_CATEGORIA_pk PRIMARY KEY (ID_CATEGORIA));

CREATE TABLE AUTOR
(ID_AUTOR NUMBER (10,0),
NOME_AUTOR VARCHAR (60),
IDADE_AUTOR NUMBER (2,0),
CONSTRAINT ID_AUTOR_pk PRIMARY KEY (ID_AUTOR));

CREATE TABLE VIDEOS
(  ID_VIDEO NUMBER(8,0),
NOME_VIDEO VARCHAR2(200),
 DUR_VIDEO NUMBER (3,0),
CONSTRAINT ID_VIDEO_pk PRIMARY KEY (ID_VIDEO));

 CREATE TABLE ASSINATURA
 (ID_ASSINATURA NUMBER (10,0),
 NOME_ASSINATURA VARCHAR (30),
 PREC_ASSINATURA NUMBER (4,0),
CONSTRAINT ID_ASSINATURA_pk PRIMARY KEY (ID_ASSINATURA));
 
 CREATE TABLE SUBCATEGORIA
 (ID_SUBCATEGORIA NUMBER (10,0),
 SUBCATEGORIA_NOME VARCHAR (50),
 ID_CATEGORIA NUMBER (10,0),
CONSTRAINT ID_SUBCATEGORIA_pk PRIMARY KEY (ID_SUBCATEGORIA),
  CONSTRAINT ID_CATEGORIA_SUBCATEGORIA
    FOREIGN KEY (ID_CATEGORIA)
    REFERENCES CATEGORIA (ID_CATEGORIA));

CREATE TABLE USUARIO
(       ID_USUARIO NUMBER(6,0),
        FIRST_USUARIO VARCHAR(50),
	LAST_USUARIO VARCHAR(50),
        EMAIL_USUARIO VARCHAR2 (200),
        CPF_USUARIO NUMBER(11,0),
        END_USUARIO VARCHAR2(200),
        ID_ASSINATURA NUMBER (10,0),
	TEL_USUARIO NUMBER(11,0),
	RG_USUARIO NUMBER (11,0),
CONSTRAINT ID_USUARIO_pk PRIMARY KEY (ID_USUARIO),
CONSTRAINT ID_ASSINATURA_USUARIO
    FOREIGN KEY (ID_ASSINATURA)
    REFERENCES ASSINATURA (ID_ASSINATURA)
);

CREATE TABLE NOTICIA 
(       ID_NOTICIA NUMBER(10,0),
        NOME_NOTICIA VARCHAR2 (400),
        ID_AUTOR NUMBER (10,0),
        ID_CATEGORIA NUMBER (10,0),
        ID_SUBCATEGORIA NUMBER (10,0),
CONSTRAINT ID_NOTICIA_pk PRIMARY KEY (ID_NOTICIA),
CONSTRAINT ID_AUTOR_NOTICIA
    FOREIGN KEY (ID_AUTOR)
    REFERENCES AUTOR (ID_AUTOR),
 CONSTRAINT ID_CATEGORIA_NOTICIA
    FOREIGN KEY (ID_CATEGORIA)
    REFERENCES CATEGORIA (ID_CATEGORIA),
CONSTRAINT ID_SUBCATEGORIA_NOTICIA
    FOREIGN KEY (ID_SUBCATEGORIA)
    REFERENCES SUBCATEGORIA (ID_SUBCATEGORIA));
 
 CREATE TABLE NOTVID
 (ID_VIDEO NUMBER(8,0),
 ID_NOTICIA NUMBER(10,0),
 CONSTRAINT ID_VIDEO_NOTVID
    FOREIGN KEY (ID_VIDEO)
    REFERENCES VIDEOS (ID_VIDEO),
 CONSTRAINT ID_NOTICIA_NOTVID
    FOREIGN KEY (ID_NOTICIA)
    REFERENCES NOTICIA (ID_NOTICIA)
);
 
 CREATE TABLE NOTAUTOR
 (ID_NOTICIA NUMBER(10,0),
 ID_AUTOR NUMBER (10,0),
 CONSTRAINT ID_NOTICIA_NOTAUTOR
    FOREIGN KEY (ID_NOTICIA)
    REFERENCES NOTICIA (ID_NOTICIA),
 CONSTRAINT ID_AUTOR_NOTAUTOR
    FOREIGN KEY (ID_AUTOR)
    REFERENCES AUTOR (ID_AUTOR));
 
 CREATE TABLE NOTASSINATURA
 (ID_NOTICIA NUMBER(10,0),
 ID_ASSINATURA NUMBER (10,0),
 CONSTRAINT ID_NOTICIA_NOTASSINATURA
    FOREIGN KEY (ID_NOTICIA)
    REFERENCES NOTICIA (ID_NOTICIA),
 CONSTRAINT ID_ASSINATURA_NOTASSINATURA
    FOREIGN KEY (ID_ASSINATURA)
    REFERENCES ASSINATURA (ID_ASSINATURA));
